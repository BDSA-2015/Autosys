				\item \parbox[t]{\linewidth-\itemindent}{ The \serverside exports the requested research protocol to the \client.}
				}

			\newline
	\end{enumerate} \\

	\vspace{-3mm}

	\vspace{-3mm}
	\begin{itemize}[leftmargin=*,topsep=0pt,itemsep=-1ex]

	\vspace{-3mm}
